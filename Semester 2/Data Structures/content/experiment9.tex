\section{Factorial}
\label{sec:Factorial}

\subsection{Objective}
Write a program for Factorial

\subsection{Algorithm}
\begin{lstlisting}[style=mystyle]
factorial(n)
    If n == 1 then return 1
    Else 
        f = n * factorial(n - 1)
    Return f
\end{lstlisting}
